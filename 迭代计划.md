```python
import zipfile
import io

async def create_batch_download_zip(image_ids: List[str]) -> bytes:
    """创建批量下载ZIP包"""
    zip_buffer = io.BytesIO()
    
    with zipfile.ZipFile(zip_buffer, 'w', zipfile.ZIP_DEFLATED) as zipf:
        for image_id in image_ids:
            image_path = self.get_image_path(image_id)
            if image_path:
                # 使用文件名作为ZIP内文件名
                metadata = self.get_image_info(image_id)
                zipf.write(image_path, metadata['filename'])
    
    zip_buffer.seek(0)
    return zip_buffer.read()
```

```typescript
// hooks/useOnboarding.ts
export const useOnboarding = () => {
  const [currentStep, setCurrentStep] = useState(0);
  const [visible, setVisible] = useState(false);
  const completedSteps = useLocalStorage('onboardingCompleted', [] as string[]);
  
  const startOnboarding = () => {
    setVisible(true);
    setCurrentStep(0);
  };
  
  const completeStep = (stepKey: string) => {
    const newCompleted = [...completedSteps.value, stepKey];
    completedSteps.set(newCompleted);
    
    if (currentStep < onboardingSteps.length - 1) {
      setCurrentStep(currentStep + 1);
    } else {
      setVisible(false);
    }
  };
  
  const skipOnboarding = () => {
    setVisible(false);
  };
  
  return {
    visible,
    currentStep,
    startOnboarding,
    completeStep,
    skipOnboarding,
    isCompleted: (stepKey: string) => completedSteps.value.includes(stepKey)
  };
};
```

```typescript
// components/common/ErrorBoundary.tsx
class ErrorBoundary extends React.Component<{ children: React.ReactNode }, { hasError: boolean; error?: Error }> {
  constructor(props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    // 上报错误
    logErrorToService(error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return <ErrorFallback error={this.state.error} />;
    }
    return this.props.children;
  }
}
```

```python
# 使用 CLIP 模型进行零样本分类
from transformers import CLIPProcessor, CLIPModel

class ImageClassifier:
    def __init__(self):
        self.processor = CLIPProcessor.from_pretrained("openai/clip-vit-base-patch32")
        self.model = CLIPModel.from_pretrained("openai/clip-vit-base-patch32")
        
        # 预定义类别
        self.categories = {
            'scene': ['landscape', 'portrait', 'food', 'travel', 'indoor', 'outdoor'],
            'object': ['cat', 'dog', 'car', 'flower', 'building', 'sky'],
            'color': ['red', 'blue', 'green', 'yellow', 'black', 'white']
        }
    
    def classify(self, image: Image.Image) -> Dict[str, List[str]]:
        """分类图片"""
        inputs = self.processor(text=self.categories, images=image, return_tensors="pt", padding=True)
        outputs = self.model(**inputs)
        logits_per_image = outputs.logits_per_image
        probs = logits_per_image.softmax(dim=-1)
        
        # 返回Top-3类别
        results = {}
        for category, labels in self.categories.items():
            top_indices = probs[0][:len(labels)].topk(3).indices
            results[category] = [labels[i] for i in top_indices]
        
        return results
```

```python
from facenet_pytorch import InceptionResnetV1, MTCNN

class FaceDetector:
    def __init__(self):
        self.mtcnn = MTCNN(keep_all=True)
        self.resnet = InceptionResnetV1(pretrained='vggface2').eval()
    
    def detect_faces(self, image: Image.Image) -> List[Dict]:
        """检测人脸并生成embedding"""
        # 检测人脸
        boxes, probs = self.mtcnn.detect(image)
        
        faces = []
        if boxes is not None:
            for i, box in enumerate(boxes):
                # 提取人脸
                face = self.mtcnn.extract_face(image, box, save_path=None)
                
                # 生成embedding
                with torch.no_grad():
                    embedding = self.resnet(face.unsqueeze(0))
                
                faces.append({
                    'box': box.tolist(),
                    'confidence': float(probs[i]),
                    'embedding': embedding.squeeze().tolist()
                })
        
        return faces
```

- **人脸聚类**：
```python
from sklearn.cluster import DBSCAN

class FaceCluster:
    def cluster_faces(self, faces: List[Dict], eps: float = 0.5) -> List[List[Dict]]:
        """人脸聚类"""
        embeddings = [f['embedding'] for f in faces]
        
        clustering = DBSCAN(eps=eps, min_samples=2).fit(embeddings)
        labels = clustering.labels_
        
        # 按聚类分组
        clusters = {}
        for face, label in zip(faces, labels):
            if label == -1:  # 噪声点
                continue
            if label not in clusters:
                clusters[label] = []
            clusters[label].append(face)
        
        return list(clusters.values())
```
